# 线程封装
```c++
class Thread {
public:
    typedef std::shared_ptr<Thread> ptr;
    // 构造函数2个参数：std::function函数对象,线程名称
    Thread(std::function<void()> cb, const std::string& name);
    ~Thread();

    pid_t getId() const { return m_id;}
    const std::string& getName() const { return m_name;}
	//封装pthread_join()函数
    void join();
	// 返回当前线程的智能指针
    static Thread* GetThis();
    static const std::string& GetName();
    static void SetName(const std::string& name);
private:
	// 禁用拷贝构造，拷贝复制和移动构造函数
    Thread(const Thread&) = delete;
    Thread(const Thread&&) = delete;
    Thread& operator=(const Thread&) = delete;
	// pthread_create()的函数的参数
    static void* run(void* arg);
private:
	pid_t m_id = -1;             //线程id，和top命令显示的线程id一样
    pthread_t m_thread = 0;      //pthread_create返回的线程标识
    std::function<void()> m_cb;  //线程运行的函数
	std::string m_name;          //线程名称

    Semaphore m_semaphore;
};
```

## 实现细节
### Thread构造函数
```c++
Thread::Thread(std::function<void()> cb, const std::string& name)
    :m_cb(cb)
    ,m_name(name) {
    if(name.empty()) {
        m_name = "UNKNOW";
    }
    int rt = pthread_create(&m_thread, nullptr, &Thread::run, this);
    if(rt) {
        SYLAR_LOG_ERROR(g_logger) << "pthread_create thread fail, rt=" << rt
            << " name=" << name;
        throw std::logic_error("pthread_create error");
    }
    // 用一个信号量等待，保证线程确实开始执行了
    m_semaphore.wait();
}
```
- 调用pthread_create创建线程 [[pthread线程相关API#pthread_create函数]]
- 等待在一个信号量上，保证线程确实开始执行了

### 线程要运行的函数run()
```c++
void* Thread::run(void* arg) {
	// arg是pthread_create传入的this指针
    Thread* thread = (Thread*)arg;
    //线程局部变量，存储该线程的指针和名字
    t_thread = thread;
    t_thread_name = thread->m_name;
    //设置该线程的id和名字
    thread->m_id = sylar::GetThreadId();
    pthread_setname_np(pthread_self(), thread->m_name.substr(0, 15).c_str());

    std::function<void()> cb;
    cb.swap(thread->m_cb);
	// 在运行用户函数前通知主线程线程确实已经开始运行了(主线程此时等待在构造函数的信号量上)
    thread->m_semaphore.notify();
    cb();
    return 0;
}
```
- run函数在运行用户传入的函数前先在线程局部变量中保存了该线程的信息，同时设置了线程id和名字等信息到该Thread对象中
- 设置线程名字调用了pthread_setname_np函数[[pthread线程相关API#pthread_setname_np函数]]

### join()函数
对pthread_join函数[[pthread线程相关API#pthread_join函数]]的简单封装
```c++
void Thread::join() {
    if(m_thread) {
        int rt = pthread_join(m_thread, nullptr);
        if(rt) {
            SYLAR_LOG_ERROR(g_logger) << "pthread_join thread fail, rt=" << rt
                << " name=" << m_name;
            throw std::logic_error("pthread_join error");
        }
        m_thread = 0;
    }
}
```

# 读写锁的封装
RWMutex是对pthread_rwlock_t的简单封装
```c++
class RWMutex {
public:
    typedef ReadScopedLockImpl<RWMutex> ReadLock;
    typedef WriteScopedLockImpl<RWMutex> WriteLock;

    RWMutex() {
        pthread_rwlock_init(&m_lock, nullptr);
    }
    
    ~RWMutex() {
        pthread_rwlock_destroy(&m_lock);
    }

    void rdlock() {
        pthread_rwlock_rdlock(&m_lock);
    }

    void wrlock() {
        pthread_rwlock_wrlock(&m_lock);
    }

    void unlock() {
        pthread_rwlock_unlock(&m_lock);
    }
private:
    pthread_rwlock_t m_lock;
};
```

另外使用RAII的思想，实现两个RWMutex的资源管理类ReadScopedLockImpl< RWMutex >和WriteScopedLockImpl< RWMutex >
```c++
template<class T>
struct ReadScopedLockImpl {
public:
// 对象构造时加锁
    ReadScopedLockImpl(T& mutex)
        :m_mutex(mutex) {
        m_mutex.rdlock();
        m_locked = true;
    }
// 对象析构时解锁
    ~ReadScopedLockImpl() {
        unlock();
    }

    void lock() {
        if(!m_locked) {
            m_mutex.rdlock();
            m_locked = true;
        }
    }

    void unlock() {
        if(m_locked) {
            m_mutex.unlock();
            m_locked = false;
        }
    }
private:
    T& m_mutex;
    bool m_locked;
};
```


# 信号量的封装
对sem_t[[pthread线程相关API#信号量]]的简单封装
```c++
class Semaphore {
public:
    Semaphore(uint32_t count = 0);
    ~Semaphore();

    void wait();
    void notify();
private:
    Semaphore(const Semaphore&) = delete;
    Semaphore(const Semaphore&&) = delete;
    Semaphore& operator=(const Semaphore&) = delete;
private:
    sem_t m_semaphore;
};

void Semaphore::wait() {
    if(sem_wait(&m_semaphore)) {
        throw std::logic_error("sem_wait error");
    }
}

void Semaphore::notify() {
    if(sem_post(&m_semaphore)) {
        throw std::logic_error("sem_post error");
    }
}
```
